34  void UpdateValue(T updatedInput)

40  T GetValue() const
41  {
42  if (auto timeSinceStartUs = GetTimeSinceStartupUs(); timeSinceStartUs - m_lastUpdateUs > m_timeoutUs)

44  return 0;
45  }
46  return m_input;
47  }
48 
49  private:

51  {
52  return static_cast<int64_t>(AZ::Interface<AZ::ITime>::Get()->GetElapsedTimeUs());
53  }
54

56  int64_t m_lastUpdateUs = 0;
57  T m_input = 0;
