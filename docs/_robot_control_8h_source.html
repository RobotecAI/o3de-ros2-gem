44  void Activate(const AZ::Entity* entity) final
45  {
46  m_active = true;
47  if (!m_controlSubscription)
48  {
49  auto ros2Frame = entity->FindComponent<ROS2FrameComponent>();
50  AZStd::string namespacedTopic = ROS2Names::GetNamespacedName(ros2Frame->GetNamespace(), m_controlConfiguration.m_topic);
51 
52  auto ros2Node = ROS2Interface::Get()->GetNode();
53  m_controlSubscription = ros2Node->create_subscription<T>(
54  namespacedTopic.data(),
55  m_controlConfiguration.m_qos.GetQoS(),
56  [this](const T& message)
57  {
58  OnControlMessage(message);
59  });
60  }
61  };
62 
63  void Deactivate() final
64  {
65  m_active = false;
66  m_controlSubscription.reset(); // Note: topic and qos can change, need to re-subscribe
67  };
68 
69  virtual ~RobotControl() = default;
70 
71  protected:
72  ControlConfiguration m_controlConfiguration;
73 
74  private:
75  void OnControlMessage(const T& message)
76  {
77  if (!m_active)
78  return;
79 
80  if (m_controlConfiguration.m_broadcastBusMode)
81  {
82  BroadcastBus(message);
83  }
84  else
85  {
86  ApplyControl(message);
87  }
88  };
89 
90  virtual void ApplyControl(const T& message) = 0;
91  virtual void BroadcastBus(const T& message) = 0;
92 
93  bool m_active = false;
